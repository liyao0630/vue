多个样式对象时可以使用数组

      

    
  
    
      new Vue({
        el: '#app',
        data: {
          isActive: true,
          isError: false,
          show: true,
          color: 'pink',
          fontSize: 20,
          styleFont: {
            fontSize: '20px'
          },
          styleColor: {
            color: 'gray'
          }
        },
        computed: {
          classesObj() {
            return {
              active: this.isActive,
              error: this.isError,
              'classesObj': true
            }
          },
          classesArr() {
            return [{
                active: this.isActive
              },
              'classesArr',
              this.show ? 'show' : ''
            ]
          },
          styleObj() {
            return {
              fontSize: this.fontSize + 'px',
              background: 'gray',
              color: this.color
            }
          }
        }
      })
